📅 18 abril 2026 🏢 ASOCOLCI · Junta Directiva 👤 Para: Ing. Andrés Suárez Soto ✍️ De: Javier Castilla Robles
🏗️ Lista para implementar — sin ambigüedad
📜 Fundamento Normativo — No negociable
Art. 24°, literal d: "Participar con voz y un (1) voto en las sesiones de la Junta Directiva."
Art. 33°: "Los miembros de la Junta de Vigilancia, con voz y un (1) voto institucional."

La JV emite 1 voto como órgano. Nunca 1 por persona. Sin excepción.
✅ SUPLENCIAS — Qué hace el motor
  • Principal presente → +1 quórum. Suplente: voz, sin voto.
  • Principal ausente + suplente presente → suplente asume. +1 quórum. Puede votar.
  • Ambos ausentes → cargo no computa.
  • 1 cargo = máximo 1 voto. Nunca ambos.
  • Sin suplencias cruzadas. Suplente de Tesorería no reemplaza Fiscalía.
❌ PROHIBIDO
  • Principal + suplente con voto simultáneo
  • Suplente cruzado en cargo diferente
  • JV contada como 2 o 3 votos individuales
  • Invitado con voto en cualquier circunstancia
  • Revisor Fiscal o Contador con voto
⚙️ JV — Los 3 casos posibles
Caso A — Presidente presente
Presidente JV actúa como representante automático. Emite el voto. Demás: solo voz.
JV = 1 voto. Automático.
Caso B — Presidente ausente
Admin designa representante de los presentes. Solo ese vota. Demás: solo voz.
JV = 1 voto. Admin designa.
Caso C — Todos ausentes
JV no tiene presencia. No computa.
JV = 0 votos.
Principio central: El eligible_voters_set se calcula UNA SOLA VEZ por sesión. Quórum, asistencia, votaciones, PDF y dashboard consumen ese set. Ningún módulo recalcula por su cuenta.

⚙️ calcularElegibilidad(session_id) — Función central
// ejecutar al registrar asistencia o actualizar presentes
function calcularElegibilidad(session_id) {
  const asistentes = getAsistentes(session_id)
  const eligible_set = []

  // 1. Suplencias — agrupar por grupo_reemplazo
  const grupos = groupBy(asistentes.filter(m => m.organo === 'JD'), 'grupo_reemplazo')
  for (const grupo of grupos) {
    const principal = grupo.find(m => m.tipo_miembro === 'principal' && m.presente)
    const suplente = grupo.find(m => m.tipo_miembro === 'suplente' && m.presente)
    if (principal) {
      eligible_set.push({ ...principal, rol_en_sesion: 'principal', computa_quorum: true, puede_votar: true })
      if (suplente) eligible_set.push({ ...suplente, rol_en_sesion: 'suplente_con_voz', computa_quorum: false, puede_votar: false })
    } else if (suplente) {
      eligible_set.push({ ...suplente, rol_en_sesion: 'suplente_actuando', computa_quorum: true, puede_votar: true })
    }
  }

  // 2. JV — 1 voto institucional
  const jvPresentes = asistentes.filter(m => m.organo === 'JV' && m.presente)
  if (jvPresentes.length > 0) {
    const rep = resolverRepresentanteJV(jvPresentes, session_id)
    if (rep) {
      eligible_set.push({ ...rep, rol_en_sesion: 'jv_representante', computa_quorum: true, puede_votar: true })
      jvPresentes.filter(m => m.member_id !== rep.member_id).forEach(m =>
        eligible_set.push({ ...m, rol_en_sesion: 'jv_voz', computa_quorum: false, puede_votar: false })
      )
    }
  }

  updateSessionMembers(session_id, eligible_set)
  return eligible_set
}
⚙️ resolverRepresentanteJV
function resolverRepresentanteJV(jvPresentes, session_id) {
  const presidente = jvPresentes.find(m => m.cargo_funcional === 'PRESIDENTE_JV')
  if (presidente) { setJVSession(session_id, { representante_jv_id: presidente.member_id, representante_designado_por: 'auto' }); return presidente }

  const jvSession = getJVSession(session_id)
  if (jvSession.representante_jv_id) return jvPresentes.find(m => m.member_id === jvSession.representante_jv_id)

  // sin presidente y sin designar → disparar UI de designación al admin
  triggerUIDesignarRepresentanteJV(session_id, jvPresentes)
  return null
}

⚙️ validarVotante(cedula, voting_id) — Formulario público
// ejecutar al recibir cédula — antes de mostrar cualquier opción
function validarVotante(cedula, voting_id) {
  const member = buscarPorCedula(cedula)
  if (!member) return { status: 'NOT_FOUND' }

  const sm = getSessionMember(member.member_id, voting_id.session_id)
  if (!sm.puede_votar) return { status: 'NO_VOTE', cargo: member.cargo_funcional }
  if (sm.rol_en_sesion === 'suplente_con_voz') return { status: 'SUPLENTE_SIN_VOTO' }
  if (sm.rol_en_sesion === 'jv_voz') return { status: 'JV_VOZ' }

  const jvSession = getJVSession(voting_id.session_id)
  if (member.organo === 'JV' && jvSession.jv_voto_emitido) return { status: 'JV_VOTED' }
  if (checkVotoExistente(member.member_id, voting_id)) return { status: 'ALREADY_VOTED' }

  return { status: 'OK', member, session_member: sm }
}

🔴 LÓGICO
Votos inflados — resultado incorrecto
Con suplentes y JV sin control, una votación con 9 elegibles puede tener 12+ votos. El ganador puede ser incorrecto. Nadie lo nota en pantalla.
🔴 JURÍDICO
Viola el estatuto — decisión impugnable
Art. 24° es explícito: 1 voto JV. Contar 3 individuales es violación estatutaria. Cualquier decisión tomada puede impugnarse.
🔴 PROBATORIO
PDF no trazable ni defendible
Sin registro de quién actuó como representante JV y qué suplente reemplazó a quién, el acta no tiene valor probatorio ante ningún conflicto.
🔴 ACTA
Sesión declarable nula
Acta donde un invitado votó, 3 JV emitieron votos individuales y principal + suplente del mismo cargo votaron: no tiene validez legal.
